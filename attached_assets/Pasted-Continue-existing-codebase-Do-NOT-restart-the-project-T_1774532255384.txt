STEP 5 — RENDERING SYSTEM
==================================================

Parse maps into 3D voxel arrays:

function parseMap(
  mapRows: string[],
  colorTable: Record<string,string|null>,
  view: 'front'|'side'|'back'
): Voxel[] {
  const voxels: Voxel[] = []
  const depth = 3  // give each 2D pixel some depth

  mapRows.forEach((row, rowIdx) => {
    const y = mapRows.length - 1 - rowIdx  // flip Y
    
    [...row].forEach((char, col) => {
      const color = colorTable[char]
      if (!color) return
      
      const centerX = row.length / 2
      const x = col - centerX
      
      // For side view: rotate 90 degrees
      if (view === 'side') {
        for (let d = 0; d < depth; d++) {
          voxels.push({ x: d - depth/2, y, z: x * 0.5, c: color })
        }
      }
      // For back view: mirror x
      else if (view === 'back') {
        for (let d = 0; d < depth; d++) {
          voxels.push({ x: -x * 0.5, y, z: d - depth/2, c: color })
        }
      }
      // Front view
      else {
        for (let d = 0; d < depth; d++) {
          voxels.push({ x: x * 0.5, y, z: d - depth/2, c: color })
        }
      }
    })
  })
  return voxels
}

Combine front + side + back into one unified 3D model:
- Front voxels: z = +2 to +4
- Side voxels: x = +5 to +8 (right side)
           and x = -5 to -8 (left side, mirrored)
- Back voxels: z = -2 to -4

This creates a hollow 3D shell that looks
correct from all angles.
