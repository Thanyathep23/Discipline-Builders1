TASK: Add a DEV MODE / ADMIN MODE toggle to the app.
When Dev Mode is ON:
  - All items (cars, watches, outfits) are unlocked regardless of level or coins
  - All level requirements are bypassed
  - A visible banner shows "DEV MODE ON" so we always know it's active
  - One tap to turn it on, one tap to turn it off
  - The toggle persists across screens (global state)

When Dev Mode is OFF:
  - Normal game rules apply (level req, coin req, locked items)

════════════════════════════════════════════════════════
STEP 1 — CREATE DEV MODE GLOBAL STATE
════════════════════════════════════════════════════════

Create a new file: utils/devMode.ts
```ts
import { useState, useEffect, createContext, useContext } from 'react'
import AsyncStorage from '@react-native-async-storage/async-storage'

const DEV_MODE_KEY = 'dev_mode_enabled'

// Global dev mode context
export const DevModeContext = createContext<{
  isDevMode: boolean
  toggleDevMode: () => void
}>({
  isDevMode: false,
  toggleDevMode: () => {},
})

export function useDevMode() {
  return useContext(DevModeContext)
}

// Hook to use in _layout.tsx provider
export function useDevModeProvider() {
  const [isDevMode, setIsDevMode] = useState(false)

  useEffect(() => {
    AsyncStorage.getItem(DEV_MODE_KEY).then(val => {
      setIsDevMode(val === 'true')
    })
  }, [])

  const toggleDevMode = async () => {
    const newVal = !isDevMode
    setIsDevMode(newVal)
    await AsyncStorage.setItem(DEV_MODE_KEY, String(newVal))
  }

  return { isDevMode, toggleDevMode }
}
```
