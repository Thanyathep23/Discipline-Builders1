```tsx
// Add TypeScript declaration
declare global {
  namespace JSX {
    interface IntrinsicElements {
      'model-viewer': React.DetailedHTMLProps<React.HTMLAttributes<HTMLElement> & {
        src?: string
        alt?: string
        'camera-controls'?: boolean | string
        'auto-rotate'?: boolean | string
        'auto-rotate-delay'?: string
        'rotation-per-second'?: string
        'camera-orbit'?: string
        'field-of-view'?: string
        'environment-image'?: string
        'shadow-intensity'?: string
        'exposure'?: string
      }, HTMLElement>
    }
  }
}

// Inject script once
useEffect(() => {
  if (typeof window === 'undefined') return
  if (!document.querySelector('[data-model-viewer-script]')) {
    const s = document.createElement('script')
    s.type = 'module'
    s.src = 'https://ajax.googleapis.com/ajax/libs/model-viewer/3.5.0/model-viewer.min.js'
    s.setAttribute('data-model-viewer-script', 'true')
    document.head.appendChild(s)
  }
}, [])

// In JSX where character should appear:
<div style={{ width: '100%', height: 320, borderRadius: 16, overflow: 'hidden' }}>
  <model-viewer
    src="/api/models/Superhero_Male_FullBody.gltf"
    camera-controls
    auto-rotate
    auto-rotate-delay="3000"
    rotation-per-second="8deg"
    camera-orbit="0deg 82deg auto"
    field-of-view="36deg"
    environment-image="neutral"
    shadow-intensity="1.2"
    exposure="1.1"
    style={{ width: '100%', height: '100%', background: 'transparent', '--poster-color': 'transparent' } as React.CSSProperties}
    alt="Character"
  />
</div>
```
